import { storeToRefs } from 'pinia'
import { useReportingStore } from '@/stores/reporting'

const reportingStore = useReportingStore()
const { loading, summaryCards, lastUpdated } = storeToRefs(reportingStore)

const palette = ['rose', 'emerald', 'amber', 'indigo', 'violet', 'sky']
const icons = ['??', '??', '??', '??', '??', '???']

const metricCards = computed(() => {
  const flattened = summaryCards.value.flatMap(card =>
    card.metrics.map(metric => ({
      key: ${card.title}-,
      title: metric.label,
      value: metric.value,
      subtitle: card.title,
    }))
  )

  if (flattened.length === 0) return []

  const sorted = [...flattened].sort((a, b) => b.value - a.value)
  return sorted.slice(0, 4).map((metric, index) => ({
    ...metric,
    variant: palette[index % palette.length],
    icon: icons[index % icons.length],
  }))
})

const metricsLoading = computed(() => loading.value.summaries)
const hasMetrics = computed(() => metricCards.value.length > 0)

const lastSyncLabel = computed(() => {
  const iso = lastUpdated.value.summaries
  if (!iso) return 'Henüz senkronize edilmedi'
  try {
    return new Intl.DateTimeFormat('tr-TR', {
      hour: '2-digit',
      minute: '2-digit',
      second: '2-digit',
    }).format(new Date(iso))
  } catch {
    return 'Tarih okunamadý'
  }
})

const refreshMetrics = async () => {
  await reportingStore.fetchSummaries()
}

onMounted(async () => {
  if (!summaryCards.value.length) {
    await reportingStore.fetchSummaries()
  }
})
